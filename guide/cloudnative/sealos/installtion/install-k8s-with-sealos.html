主机准备

配置主机名称

手动设置
# 对应的节点进行设置
hostnamectl set-hostname node1
hostnamectl set-hostname node2
hostnamectl set-hostname node3
hostnamectl set-hostname node4
hostnamectl set-hostname node5
脚本设置
# 在管理节点上执行即可
hosts="10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214"
num=1
for host in $hosts; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} hostnamectl set-hostname node$num ;num=$(($num+1));done

配置 hosts

master 节点+worker 节点

cat >> /etc/hosts <<EOF
10.10.12.210 node1
10.10.12.211 node2
10.10.12.212 node3
10.10.12.213 node4
10.10.12.214 node5
EOF

关闭 selinux 和防火墙

master 节点+worker 节点

sed -i 's/^SELINUX=enforcing/SELINUX=disabled/g' /etc/selinux/config
setenforce 0
sestatus

systemctl stop firewalld.service
systemctl disable firewalld.service
firewall-cmd --state

关闭 swap

master 节点+worker 节点

sed -i 's/.*swap.*/#&/' /etc/fstab
swapoff -a
free -m

安装依赖

yum install -y conntrack socat

时钟同步

yum install -y ntpdate
ntpdate ntp1.aliyun.com

echo "* */1 * * * ntpdate ntp1.aliyun.com" > /tmp/crontab.txt

crontab /tmp/crontab.txt

crontab -l

rm -f /tmp/crontab.txt

内核升级（可选）

通用
# Ubuntu忽略，CentOS执行
 
# 安装最新的内核
# 我这里选择的是稳定版kernel-ml   如需更新长期维护版本kernel-lt  
yum -y --enablerepo=elrepo-kernel  install  kernel-ml
 
# 查看已安装那些内核
rpm -qa | grep kernel
 
# 查看默认内核
grubby --default-kernel
 
# 若不是最新的使用命令设置
grubby --set-default $(ls /boot/vmlinuz-* | grep elrepo)
 
# 重启生效
reboot
离线
# 离线版本 
yum install -y /root/cby/kernel-lt-*-1.el7.elrepo.x86_64.rpm 
grubby --set-default $(ls /boot/vmlinuz-* | grep elrepo) 
grubby --default-kernel 
reboot
CentOS 7.x
# v7 整合命令为：
yum install https://www.elrepo.org/elrepo-release-7.el7.elrepo.noarch.rpm -y 
sed -i "s@mirrorlist@#mirrorlist@g" /etc/yum.repos.d/elrepo.repo 
sed -i "s@elrepo.org/linux@mirrors.tuna.tsinghua.edu.cn/elrepo@g" /etc/yum.repos.d/elrepo.repo 
yum  --disablerepo="*"  --enablerepo="elrepo-kernel"  list  available -y 
yum  --enablerepo=elrepo-kernel  install  kernel-lt -y 
grubby --set-default $(ls /boot/vmlinuz-* | grep elrepo) 
grubby --default-kernel 
reboot
CentOS 8.x
# v8 整合命令为：
yum install https://www.elrepo.org/elrepo-release-8.el8.elrepo.noarch.rpm -y 
sed -i "s@mirrorlist@#mirrorlist@g" /etc/yum.repos.d/elrepo.repo
sed -i "s@elrepo.org/linux@mirrors.tuna.tsinghua.edu.cn/elrepo@g" /etc/yum.repos.d/elrepo.repo
yum  --disablerepo="*"  --enablerepo="elrepo-kernel"  list  available -y
yum  --enablerepo=elrepo-kernel  install kernel-lt -y 
grubby --default-kernel
reboot

安装sealos

在管理节点上安装sealos（非k8s集群节点）

wget https://github.com/labring/sealos/releases/download/v4.3.7/sealos_4.3.7_linux_amd64.tar.gz

tar -zxvf sealos_4.3.7_linux_amd64.tar.gz sealos 
chmod +x sealos
mv sealos /usr/bin/

离线镜像准备（可选）

安装sealos

在集群节点上安装sealos二进制包

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} rm -rf /root/sealos-workspace; done

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} mkdir /root/sealos-workspace; done

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' scp -o StrictHostKeyChecking=no /root/sealos-workspace/sealos_4.3.7_linux_amd64.tar.gz root@${host}:/root/sealos-workspace/; done

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} tar -zxvf /root/sealos-workspace/sealos_4.3.7_linux_amd64.tar.gz sealos; done

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} chmod +x sealos; done

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} cp -fp sealos /usr/bin/; done

准备kuberntes镜像

在管理节点上下载镜像（非k8s集群节点）

sealos pull labring/kubernetes:v1.27.7

在管理节点打包镜像（非k8s集群节点）

sealos save -o /root/sealos-workspace/kubernetes.tar labring/kubernetes:v1.27.7

上传打包的镜像到其他节点

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' scp -o StrictHostKeyChecking=no /root/sealos-workspace/kubernetes.tar root@${host}:/root/sealos-workspace/; done

在k8s节点导入镜像

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} sealos load -i /root/sealos-workspace/kubernetes.tar;done

查看镜像导入情况

for host in 10.10.12.210 10.10.12.211 10.10.12.212 10.10.12.213 10.10.12.214; do sshpass -p 'root@123' ssh -o StrictHostKeyChecking=no root@${host} sealos images;done

部署k8s集群

单机部署

# 其中--env criData为指定cri的存储目
sealos run labring/kubernetes:v1.27.7 labring/helm:v3.12.3 labring/flannel:v0.22.1 --env criData=/home/containerd-data --single

#删除污点
kubectl taint node --all node-role.kubernetes.io/control-plane-

集群部署

calico
# 其中-p [your-ssh-passwd]为节点的密码，如果各节点密码不一样的话  可以在sealos所在的节点生成密钥，然后对其他节点做免密登录配置
sealos run labring/kubernetes:v1.27.7 labring/helm:v3.12.3 labring/calico:v3.24.6 \
     --masters 10.10.12.210,10.10.12.211,10.10.12.212 \
     --nodes 10.10.12.213 --port=22 -p root@123
calico+国内镜像
sealos run registry.cn-shanghai.aliyuncs.com/labring/kubernetes:v1.27.7 registry.cn-shanghai.aliyuncs.com/labring/helm:v3.12.3 registry.cn-shanghai.aliyuncs.com/labring/calico:v3.24.6 \
     --masters 10.10.12.210,10.10.12.211,10.10.12.212 \
     --nodes 10.10.12.213 --port=22 -p root@123
cilium
sealos run labring/kubernetes:v1.27.7 labring/helm:v3.9.4 labring/cilium:v1.13.4 \
     --masters 10.10.12.210,10.10.12.211,10.10.12.212 \
     --nodes 10.10.12.213 --port=22 -p root@123
cilium+国内镜像
sealos run registry.cn-shanghai.aliyuncs.com/labring/kubernetes:v1.27.7 registry.cn-shanghai.aliyuncs.com/labring/helm:v3.9.4 registry.cn-shanghai.aliyuncs.com/labring/cilium:v1.13.4 \
     --masters 10.10.12.210,10.10.12.211,10.10.12.212 \
     --nodes 10.10.12.213 --port=22 -p root@123
Clusterfile自定义安装

运行 sealos gen 生成一个 Clusterfile

sealos gen labring/kubernetes:v1.27.7 labring/helm:v3.12.3 labring/calico:v3.24.6 \
   --masters 10.10.12.210,10.10.12.211,10.10.12.212 \
   --nodes 10.10.12.213,10.10.12.214 --port=22 --passwd root@123 > Clusterfile

Calico 内容

---
apiVersion: kubeadm.k8s.io/v1beta2
kind: ClusterConfiguration
networking:
  podSubnet: 10.160.0.0/12
---
apiVersion: apps.sealos.io/v1beta1
kind: Config
metadata:
  name: calico
spec:
  path: manifests/calico.yaml
  data: |
    apiVersion: operator.tigera.io/v1
    kind: Installation
    metadata:
      name: default
    spec:
      # Configures Calico networking.
      calicoNetwork:
        # Note: The ipPools section cannot be modified post-install.
        ipPools:
        - blockSize: 26
          # Note: Must be the same as podCIDR
          cidr: 10.160.0.0/12
          encapsulation: IPIP
          natOutgoing: Enabled
          nodeSelector: all()
        nodeAddressAutodetectionV4:
          interface: "eth.*|en.*"

提示

将calico的内容追加到Clusterfile文件的末尾。

启动集群安装

sealos apply -f Clusterfile

提示

集群运行成功后会把 Clusterfile 保存到 .sealos/default/Clusterfile 文件中，可以修改其中字段来重新 apply 对集群进行变更。

安装应用

sealos run labring/helm:v3.9.4

sealos run labring/openebs:v1.9.0
